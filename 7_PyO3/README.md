### –ö—Ä–æ–∫ 5: –•–µ—à—É–≤–∞–Ω–Ω—è –ø–∞—Ä–æ–ª—é

```rust
    password_hash::{rand_core::OsRng, PasswordHash, PasswordHasher, PasswordVerifier, SaltString},
    Argon2,
};
use pyo3::prelude::*;

#[pyfunction]
fn hash_password(password: &str) -> PyResult<String> {
    let salt = SaltString::generate(&mut OsRng);
    // Argon2 –∑ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏ –∑–∞ –∑–∞–º–æ–≤—á—É–≤–∞–Ω–Ω—è–º (Argon2id v19)

    // –•–µ—à-–ø–∞—Ä–æ–ª—å –¥–æ —Ä—è–¥–∫–∞ PHC ($argon2id$v=19$...)
    let password_hash = argon2
        .hash_password(password.as_bytes(), &salt)
        .map_err(|e| pyo3::exceptions::PyValueError::new_err(e.to_string()))?
        .to_string();
    Ok(password_hash)
}

#[pyfunction]
fn verify_password(password: &str, password_hash: &str) -> PyResult<bool> {
    // –ó–≤—ñ—Ä–∏—Ç–∏ –ø–∞—Ä–æ–ª—å –∑ PHC-—Ä—è–¥–∫–æ–º.
    let parsed_hash = PasswordHash::new(password_hash)
        .map_err(|e| pyo3::exceptions::PyValueError::new_err(e.to_string()))?;
    let is_valid = Argon2::default()
        .verify_password(password.as_bytes(), &parsed_hash)
        .is_ok();

    Ok(is_valid)
}
/// Python –º–æ–¥—É–ª—å –Ω–∞ –æ—Å–Ω–æ–≤—ñ Rust
#[pymodule]
fn password_hasher(m: &Bound<'_, PyModule>) -> PyResult<()> {
    m.add_function(wrap_pyfunction!(hash_password, m)?)?;
    m.add_function(wrap_pyfunction!(verify_password, m)?)?;
    Ok(())
}

```
